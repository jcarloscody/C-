git --version             



git config -global user.name ""               // Git saber quem você é
git config -global user.email ""




git init                  //inicializa o git. criou o repositório



git status                   //vai deixar o git ciente dos arquivos não comitados e comitados


git add name_file                 //adicionar o arquivo ao ambiente de teste "Staging Environment"
git add --all  ou  git add -A          //para todos


git commit -m "mensagem sobre a mudança"



git  log              //ver o histórico de um repositorio


git help --all - Veja todos os comandos possíveis




git branch nome_branch              //criar ramificacao




git branch           //ver as ramificacoes



git checkout nome_branch              //muda de branch

git checkout -b novo_branch               //cria um novo branch local

git push origin novo_branch		//depois de comitado pode dar um push para enviar o novo brach


git pull                            //atualizar

git push origin nome_da_origem                //manda para o branch  de origem



git update-index --assume-unchanged .environment                   //ignorar o environment no git add e git commit


git fetch                                                  //obtém todo o histórico de alterações de um branch / repo rastreado.

git merge nome_bbran                                             // combina a ramificação atual com uma ramificação especificada

git branch -a               //ver todas as branch filiais locais e remotas:


git branch -r                   //é apenas para branch filiais remotas.




git clone url      //clonar
